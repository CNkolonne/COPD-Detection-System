    # Too Short: Add zeros (Padding)

import os

classes = ["non-copd", "copd"]

data = []

DATASET_PATH = '/content/drive/MyDrive/Highflyers_COPD_Colab/cough_dataset'

print(f"Starting processing for classes: {classes}...")

for i, label in enumerate(classes):
    path = os.path.join(DATASET_PATH, label)
    print(f"Processing folder: {label}...")

    # Check if folder exists before entering loop
    if not os.path.exists(path):
        print(f"ERROR: Folder not found: {path}")
        continue

    for file in os.listdir(path):
        if file.endswith(".wav"):
            file_path = os.path.join(path, file)
            try:
                # 1. Load & Resample
                y, sr = librosa.load(file_path, sr=SAMPLE_RATE)

                # 2. Preprocess (Normalize -> Filter -> Trim -> Pad)
                y = librosa.util.normalize(y)
